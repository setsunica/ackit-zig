const std = @import("std");

fn solve(
    allocator: std.mem.Allocator,
    scanner: *io.Scanner(io.StdinReader),
    printer: *io.Printer(io.StdoutWriter),
) !void {
    _ = allocator;
    const parsed = try scanner.scan([]const u8);
    defer parsed.deinit();
    const input = parsed.value;
    try printer.print(input);
}

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    try io.interactStdio(gpa.allocator(), solve);
}
